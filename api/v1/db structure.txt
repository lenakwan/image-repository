CREATE TABLE users(
    user_name varchar(40) NOT NULL,
    user_pw varchar(40) NOT NULL,
    user_id SERIAL PRIMARY KEY,
    admin boolean
);
CREATE TABLE items (
    item_name varchar(40) NOT NULL,
    image TEXT NOT NULL,
    item_category varchar(40) NOT NULL,
    list_date date NOT NULL,
    quantity INT,
    private boolean,
    price NUMERIC,
    discount INT,
    img_id SERIAL PRIMARY KEY,
    fk_user_id integer
);
